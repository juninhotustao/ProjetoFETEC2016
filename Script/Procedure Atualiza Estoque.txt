CREATE PROCEDURE ATUALIZA_ESTOQUE (
	@MOVIMENTO CHAR(1),
	@QTDE DECIMAL(10,2),
	@PRO_ID INT
) AS	
BEGIN
	IF @MOVIMENTO = 'E'
		UPDATE PRODUTO SET PRO_ESTOQUE = (PRO_ESTOQUE + @QTDE) WHERE PRO_ID = @PRO_ID
	
	IF @MOVIMENTO = 'S'	
		UPDATE PRODUTO SET PRO_ESTOQUE = (PRO_ESTOQUE - @QTDE) WHERE PRO_ID = @PRO_ID	
END
