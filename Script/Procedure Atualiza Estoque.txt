CREATE PROCEDURE ATUALIZA_ESTOQUE (
	@MOVIMENTO CHAR(1),
	@QTDE DECIMAL(10,2),
	@PRO_REFERENCIA VARCHAR(20)
) AS	
BEGIN
	IF @MOVIMENTO = 'E'
		UPDATE PRODUTO SET PRO_ESTOQUE = (PRO_ESTOQUE + @QTDE) WHERE PRO_REFERENCIA = @PRO_REFERENCIA
	
	IF @MOVIMENTO = 'S'	
		UPDATE PRODUTO SET PRO_ESTOQUE = (PRO_ESTOQUE - @QTDE) WHERE PRO_REFERENCIA = @PRO_REFERENCIA
END
