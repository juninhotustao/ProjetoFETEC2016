CREATE TABLE VENDA(
	VEN_ID INT IDENTITY(1, 1),
	VEN_DESCONTO DECIMAL(12,2) DEFAULT 0,
	VEN_DATA DATETIME,	
	VEN_OBSERVACAO VARCHAR(70),
	VEN_TOTAL DECIMAL(12,2) DEFAULT 0,
		
	PRIMARY KEY (VEN_ID)	
)

CREATE TABLE ITEM_VENDA(
	ITE_ID INT IDENTITY(1, 1),
	VEN_ID INT,
	PRO_ID INT NOT NULL, 
	ITE_REFERENCIA VARCHAR(9) NOT NULL,
	ITE_PRO_DESCRICAO VARCHAR(50) NOT NULL,
	ITE_PRECO_VENDA DECIMAL(12,2) NOT NULL DEFAULT 0,
	ITE_PRECO_CUSTO DECIMAL(12,2) NOT NULL DEFAULT 0,
	ITE_QTDE DECIMAL(12,2) NOT NULL DEFAULT 0,
	ITE_TOTAL DECIMAL(12,2) NOT NULL DEFAULT 0,
	
	PRIMARY KEY (ITE_ID),
	CONSTRAINT FK_ITEM_VENDA_VEN_ID FOREIGN KEY (VEN_ID) REFERENCES VENDA (VEN_ID)
		ON DELETE CASCADE	
)
