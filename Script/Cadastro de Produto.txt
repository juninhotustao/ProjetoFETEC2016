CREATE TABLE PRODUTO(
	PRO_ID INT IDENTITY(1, 1), 
	PRO_COD_BARRAS VARCHAR(13),
	PRO_REFERENCIA VARCHAR(9) NOT NULL,
	PRO_DESCRICAO VARCHAR(50) NOT NULL,
	PRO_ESTOQUE DECIMAL(12,2) NOT NULL DEFAULT 0,
	PRO_PRECO_VENDA DECIMAL(12,2) NOT NULL DEFAULT 0,
	PRO_PRECO_CUSTO DECIMAL(12,2) NOT NULL DEFAULT 0,
	PRO_ESTOQUE_MAX INTEGER,
	
	PRIMARY KEY (PRO_ID)
)
